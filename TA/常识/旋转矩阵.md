// Rotate around X axis

float3x3  RotationX(float angle)

{

float c = cos(angle);

float s = sin(angle);

return  float3x3(

1, 0, 0,

0, c,-s,

0, s, c

);

}

  

// Rotate around Y axis

float3x3  RotationY(float angle)

{

float c = cos(angle);

float s = sin(angle);

return  float3x3(

c, 0, s,

0, 1, 0,

-s, 0, c

);

}

  

// Rotate around Z axis

float3x3  RotationZ(float angle)

{

float c = cos(angle);

float s = sin(angle);

return  float3x3(

c,-s, 0,

s, c, 0,

0, 0, 1

);

}
