#设置一个自定义的rediscli客户端：

    docker  exec  -it  redis  sh  -c  "redis-cli --tls \
    
    --cert /etc/ssl/redis/redis.crt \
    
    --key /etc/ssl/redis/redis.key \
    
    --cacert /etc/ssl/redis/ca.crt \
    
    -h 127.0.0.1 -p 6379 \
    
    --user mainSubscriberYPWS \
    
    --pass 'ypwsCYF95733714!Test%*SubUserYaKnowHaHa' \
    
    PSUBSCRIBE '__keyevent@*__:*' 'im:*'"

#
设置一个启动项：

    docker  exec  -it  redis  sh  -c  "redis-cli --tls \

    --cert /etc/ssl/redis/redis.crt \
    
    --key /etc/ssl/redis/redis.key \
    
    --cacert /etc/ssl/redis/ca.crt \
    
    -h 127.0.0.1 -p 6379 \
    
    --user mainUserYPWS \
    
    --pass 'ypwsCYF95733714!Test%*MainUserYouKnowHaHa' \
    
    SET test:expire:demo demo EX 5"
